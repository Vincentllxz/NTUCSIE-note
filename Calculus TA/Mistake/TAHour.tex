\begin{problem}
Find the following the indefinite integral
\[
	\int x (\ln x)^2 \ \text{d} x
\]
\end{problem}
\vspace{-12pt}
\begin{cor}
	\[ 
		\frac{x^2 (\ln x)^2}{2} - x^2 \ln x + \frac{x^2}{4} + C
	\]
\end{cor}


\begin{reason}
	See the following steps:
	\begin{enumerate}[label=$\arabic*^\circ$]
		\item let $u = \ln x \ (\displaystyle \text{d} x = \frac{1}{x} \text{d} u)$, then do the substitution to get \[
			\int x (\ln x)^2 \ \text{d} x = \int e^u u^2 (x \ \text{d} u) = \int u^2 e^{2u} \ \text{d} u
		\]
		\item Use integration by parts twice:
		\[
		\begin{tabular}{c|cc}
			 & D & I \\
			\hline
			+ & $u^2$ & $e^{2u}$ \\
			- & $2u$ & $\frac{1}{2} e^{2u}$ \\
			+ & $2$ & $\frac{1}{4} e^{2u}$ \\
		\end{tabular}
		\]
		we have \begin{align*}
			\int u^2 e^{2u} \ \text{d} u &= \frac{1}{2} u^2 e^{2u} - \frac{1}{2} ue^{2u} + \frac{1}{2} \int e^{2u} \ \text{d} u \\
			&= \frac{1}{2} u^2 e^{2u} - \frac{1}{2} ue^{2u} + \frac{1}{4} e^{2u} + C \\
			&= \frac{x^2 (\ln x)^2}{2} - x^2 \ln x + \frac{x^2}{4} + C 
		\end{align*}
	\end{enumerate}
\end{reason}

\begin{problem}
Find the following the indefinite integral
\[
	\int \frac{5}{[(ax)^2 + b^2]^{3/2}} \text{d} x
\]
\end{problem}
\vspace{-12pt}
\begin{cor}
	\[ 
		-\frac{5}{a b \sqrt{b^2 - a^2 x^2}} + C
	\]
\end{cor}

\begin{reason}
	We have to convert it into a standard form \[
		\sec^2 \theta = 1 + \tan^2 \theta
	\]
	So we let $\displaystyle x = \frac{b}{a} \sec \theta$
	and we get \begin{align*}
		\int \frac{5}{[(ax)^2 + b^2]^{3/2}} \text{d} x &= \int \frac{5}{[a^2 (\frac{b}{a} \sec \theta)^2 + b^2]^{3/2}} \cdot \frac{b}{a} \sec \theta \tan \theta \ \text{d} \theta \\
		&= \int \frac{5}{[b^2 \sec^2 \theta + b^2]^{3/2}} \cdot \frac{b}{a} \sec \theta \tan \theta \ \text{d} \theta \\
		&= \int \frac{5}{b^3 (\sec^2 \theta)^{3/2}} \cdot \frac{b}{a} \sec \theta \tan \theta \ \text{d} \theta \\
		&= \frac{5}{a b^2} \int \frac{\cos \theta}{\sin^2 \theta} \ \text{d} \theta
	\end{align*}
	then, let $u = \sin \theta \ (\text{d} \theta = \frac{1}{\cos \theta} \text{d} u)$, we have \begin{align*}
		\frac{5}{a b^2} \int \frac{\cos \theta}{\sin^2 \theta} \ \text{d} \theta &= \frac{5}{a b^2} \int \frac{\cos \theta}{u^2} \cdot \frac{1}{\cos \theta} \text{d} u \\
		&= \frac{5}{a b^2} \int u^{-2} \text{d} u \\
		&= \frac{5}{a b^2} \cdot (-u^{-1}) + C \\
		&= -\frac{5}{a b^2 \sin \theta} + C
	\end{align*}
	Finally, we have to convert back to $x$: \[
		\sin \theta = \sqrt{1 - \cos^2 \theta} = \sqrt{1 - \left( \frac{a x}{b} \right)^2} = \frac{\sqrt{b^2 - a^2 x^2}}{b}
	\]
	Thus, the final answer is \[
		-\frac{5}{a b^2} \cdot \frac{b}{\sqrt{b^2 - a^2 x^2}} + C = -\frac{5}{a b \sqrt{b^2 - a^2 x^2}} + C
	\]
\end{reason}

\begin{problem}
	Find the following the indefinite integral
	\[
		\int -6 \ln (x^2 - 1) \ \text{d} x
	\]
\end{problem}
\vspace{-12pt}
\begin{cor}
	\[ 
		?
	\]
\end{cor}

\begin{reason}
	Do the factorization first: \[
		\int -6 \ln (x^2 - 1) \ \text{d} x = \int -6 \ln (x - 1) \ \text{d} x + \int -6 \ln (x + 1) \ \text{d} x
	\]
	Then, we can do the integration by parts for each part:
	\begin{enumerate}[label=$\arabic*^\circ$]
		\item for $\displaystyle \int -6 \ln (x - 1) \ \text{d} x$:
		\[
		\begin{tabular}{c|cc}
			 & D & I \\
			\hline
			+ & $-6 \ln (x - 1)$ & $1$ \\
			- & $-\frac{6}{x - 1}$ & $x$ \\
		\end{tabular}
		\]
		we have \begin{align*}
			\int -6 \ln (x - 1) \ \text{d} x &= -6 x \ln (x - 1) + 6 \int \frac{x}{x - 1} \ \text{d} x \\
			&= -6 x \ln (x - 1) + 6 \int \left(1 + \frac{1}{x - 1} \right) \text{d} x \\
			&= -6 x \ln (x - 1) + 6 x + 6 \ln |x - 1| + C
		\end{align*}
		\item for $\displaystyle \int -6 \ln (x + 1) \ \text{d} x$:
		\[
		\begin{tabular}{c|cc}
			 & D & I \\
			\hline
			+ & $-6 \ln (x + 1)$ & $1$ \\
			- & $-\frac{6}{x + 1}$ & $x$ \\
		\end{tabular}
		\]
		we have \begin{align*}
			\int -6 \ln (x + 1) \ \text{d} x &= -6 x \ln (x + 1) + 6 \int \frac{x}{x + 1} \ \text{d} x \\
			&= -6 x \ln (x + 1) + 6 \int \left(1 - \frac{1}{x + 1} \right) \text{d} x \\
			&= -6 x \ln (x + 1) + 6 x - 6 \ln |x + 1| + C
		\end{align*}
	\end{enumerate}
\end{reason}

\begin{problem}
	Find the following the indefinite integral
	\[
		\int x^2 \arctan (5x) \ \text{d} x
	\]
\end{problem}
\vspace{-12pt}
\begin{cor}
	\[ 
		\frac{x^3}{3} \arctan (5x) - \frac{5}{3} \cdot \frac{1}{1250} \left( 1 + 25 x^2 - \ln |1 + 25 x^2| \right) + C
	\]
\end{cor}

\begin{reason}
	Doing the integration by parts follow the LIATE rule:
	\[
		\begin{tabular}{c|cc}
			 & D & I \\
			\hline
			+ & $\arctan (5x)$ & $x^2$ \\
			- & $\frac{5}{1 + 25 x^2}$ & $\frac{x^3}{3}$ \\
		\end{tabular}
	\]
	We get \[
		\int x^2 \arctan (5x) \ \text{d} x = \frac{x^3}{3} \arctan (5x) - \frac{5}{3} \int \frac{x^3}{1 + 25 x^2} \ \text{d} x
	\]
	Use substitution $u = 1 + 25 x^2$ to solve the remaining integral.
	\begin{align*}
		\int \frac{x^3}{1 + 25 x^2} \ \text{d} x = \int \frac{x^3}{u} \cdot \frac{\text{d} u}{50 x} &= \frac{1}{50} \int \frac{(u - 1)/25}{u} \text{d} u \\ &= \frac{1}{1250} \left( u - \ln |u| \right) + C
	\end{align*}
	Thus, the final answer is \[
		\frac{x^3}{3} \arctan (5x) - \frac{5}{3} \cdot \frac{1}{1250} \left( 1 + 25 x^2 - \ln |1 + 25 x^2| \right) + C
	\]
\end{reason}

\begin{problem}
	Let $\displaystyle I_n = \int sec^n(x) \ \text{d} x$. Prove that \[
		I_n = \frac{1}{n - 1} \tan x \sec^{n - 2} (x) + \frac{n - 2}{n - 1} I_{n - 2}
	\]
\end{problem}

\begin{reason}
	First imply the integration by parts:
	\[
		\begin{tabular}{c|cc}
			 & D & I \\
			\hline
			+ & $\sec^{n - 2} (x)$ & $\sec x \tan x$ \\
			- & $(n - 2) \sec^{n - 3} (x) \sec x \tan x$ & $\sec x$ \\
		\end{tabular}
	\]
	Then we get \begin{align*}
	I_n 
	&= \int \sec^{\,n-2}(x)\,\sec^{2}(x)\,dx \\[6pt]
	&= \tan(x)\sec^{\,n-2}(x) 
		- (n-2)\int \sec^{\,n-2}(x)\tan^{2}(x)\,dx \\[6pt]
	&= \tan(x)\sec^{\,n-2}(x) 
		- (n-2)\int \sec^{\,n-2}(x)(\sec^{2}x - 1)\,dx \\[6pt]
	&= \tan(x)\sec^{\,n-2}(x)
		- (n-2)\int \underset{I_n}{\underbrace{\sec^{n}(x)}}\,dx
		+ (n-2)\int \underset{I_{n-2}}{\underbrace{\sec^{\,n-2}(x)}}\,dx \\[6pt]
	&= \tan(x)\sec^{\,n-2}(x)
		- (n-2)I_n
		+ (n-2)I_{n-2}
	\end{align*}

	Rearranging the equation, we have \[
		I_n = \frac{1}{n - 1} \tan x \sec^{n - 2} (x) + \frac{n - 2}{n - 1} I_{n - 2}
	\]
\end{reason}
